#### 문제 풀이

### c++ 풀이
```c++
class Solution {
public:
    vector<string> fizzBuzz(int n) {
        vector<string> answer(n);
        for(int i=1; i<=n; ++i){
            bool divBy3 = i % 3 == 0;
            bool divBy5 = i % 5 == 0;            
            if(divBy3&&divBy5) answer[i-1]="FizzBuzz";
            else if(divBy3) answer[i-1]="Fizz";
            else if(divBy5) answer[i-1]="Buzz";
            else answer[i-1]=to_string(i);
        }
        return answer;
    }
};
```

### 다른 사람 풀이
```c++
class Solution {
public:
    vector<string> fizzBuzz(int n) {
        vector<string> answer;
        for(int i=1; i<=n; ++i){
            string s;
            bool divBy3 = i % 3 == 0;
            bool divBy5 = i % 5 == 0;

            if(divBy3) s+="Fizz";
            if(divBy5) s+="Buzz";
            if(s.empty()) s+=to_string(i);
            
            answer.push_back(s);
        }
        return answer;
    }
};
```
