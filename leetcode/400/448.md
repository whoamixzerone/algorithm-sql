#### 문제 풀이

### c++ 풀이
- Runtime: 143ms
- Beats: 5.22%
```c++
class Solution {
public:
    vector<int> findDisappearedNumbers(vector<int>& nums) {
        int cnt=0, n=nums.size();
        map<int, int> m;
        for(int i : nums) m[i]++;        
        for(int i=1; i<=n; ++i) {
            if(!m[i]) {
                nums[cnt++]=i;
            }
        }
        nums.erase(nums.begin()+cnt, nums.end());
        return nums;
    }
};
```
