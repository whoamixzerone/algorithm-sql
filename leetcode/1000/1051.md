#### 문제 풀이

### c++ 풀이
- Runtime: 0ms
- Beats: 100.00%
```c++
class Solution {
public:
    int heightChecker(vector<int>& heights) {
        int n=heights.size();
        int ans=0;
        vector<int> v=heights;
        sort(v.begin(), v.end());        
        for(int i=0; i<n; ++i) {
            if(heights[i]!=v[i]) ans++;
        }
        return ans;
    }
};
```

class Solution {

        int count = 0;
        for (size_t i = 0; i < heights.size(); ++i) {

        }

    }
