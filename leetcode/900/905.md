#### 문제 풀이

### c++ 풀이
- Runtime: 0ms
- Beats: 100.00%
```c++
class Solution {
public:
    vector<int> sortArrayByParity(vector<int>& nums) {
        int n=nums.size();
        for(int i=0, j=0; i<n; ++i) {
            if(nums[i]%2==0) swap(nums[j++], nums[i]);
        }
        return nums;
    }
};
```

class Solution {
public:
    vector<int> sortArrayByParity(vector<int>& nums) {

        int i = 0, j=nums.size()-1;
        for (i=0,j=nums.size()-1 ; i<= j ; )
        {
            if (nums[i]%2!=0 && nums[j]%2 == 0)
            {
                swap (nums[i],nums[j]);
                i++;
                j--;
            }
            if (nums[i]%2==0)
            {
                i++;
            }
            if(nums[j]%2!=0)
            {
                j--;
            }
        }

        return nums;
        
    }
};
