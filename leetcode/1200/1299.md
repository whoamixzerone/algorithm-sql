### c++ 풀이
- Runtime: 2292ms
- Beats: 5.52%
```c++
class Solution {
public:
    vector<int> replaceElements(vector<int>& arr) {
        for(int i=0; i<arr.size()-1; ++i) {
            arr[i]=*max_element(arr.begin()+i+1, arr.end());
        }
        arr[arr.size()-1]=-1;
        return arr;
    }
};
```

class Solution {
public:
    vector<int> replaceElements(vector<int>& arr) {
        int n  = arr.size() ; 
        int maxi  = -1;

        for(int i = n-1 ; i>=0 ; i--){
            int ele = arr[i];
            arr[i] = maxi;
            maxi = max(maxi,ele);
        }
        return arr;
    }
};
