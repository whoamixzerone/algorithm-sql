#### 문제 풀이

### c++ 풀이
```c++
class Solution {
public:
    int findNumbers(vector<int>& nums) {
        int answer=0;
        for(int n : nums) {
            int digit=0;
            while(n) {
                n/=10;
                digit++;
            }
            if(digit%2==0) answer++;
        }
        return answer;
    }
};
```
