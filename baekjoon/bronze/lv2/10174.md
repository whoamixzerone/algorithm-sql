#include <bits/stdc++.h>

int n;
string s;
int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    read(n);
    cin.ignore();
    while(n--) {
        getline(cin, s);
        transform(all(s), s.begin(), ::tolower);
        string rs;
        FOR(i, sz(s)-1, sz(s)&1?sz(s)/2:sz(s)/2-1, -1)
            rs+=s[i];        
        print(s.substr(0, sz(s)/2)==rs?"Yes":"No");
    }
    return 0;
}
