### c++ 풀이
```c++
#include <string>
#include <vector>
#include <stack>
using namespace std;

int solution(int n, int w, int num) {
    int answer = 0, idx=0, pos=0;
    bool flag=true;
    vector<stack<int>> v(w);
    for(int i=1; i<=n; ++i){
        if(i==num) pos=idx;
        if(flag) v[idx++].push(i);
        else v[idx--].push(i);
        if(!(i%w)){
            flag=!flag;
            !flag?idx--:idx++;
        }
    }
    while(!v[pos].empty()){
        answer++;
        int x=v[pos].top();
        v[pos].pop();
        if(x==num) break;
    }
    return answer;
}
```
